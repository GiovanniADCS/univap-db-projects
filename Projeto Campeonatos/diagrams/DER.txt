[Modalidade]
  PK id_modalidade
  nome
  tipo_resultado
  descricao
  ativo

[Categoria]
  PK id_categoria
  nome
  idade_minima
  idade_maxima
  descricao
  ativo

[Genero]
  PK id_genero
  sigla
  nome
  ativo

[Temporada]
  PK id_temporada
  ano
  nome
  data_inicio
  data_fim
  situacao

[Campeonato]
  PK id_campeonato
  FK id_modalidade -> Modalidade.id_modalidade
  FK id_categoria -> Categoria.id_categoria
  FK id_genero -> Genero.id_genero
  FK id_temporada -> Temporada.id_temporada
  nome
  formato
  situacao
  data_inicio
  data_fim
  descricao

[Equipe]
  PK id_equipe
  nome
  cidade
  estado
  pais
  data_fundacao
  ativo

[Equipe_Campeonato]
  PK id_equipe_campeonato
  FK id_equipe -> Equipe.id_equipe
  FK id_campeonato -> Campeonato.id_campeonato
  grupo
  numero_inscricao
  quantidade_atletas
  data_inscricao
  situacao

[Atleta]
  PK id_atleta
  FK id_genero -> Genero.id_genero
  nome
  data_nascimento
  nacionalidade
  documento
  ativo

[Atleta_Equipe]
  PK id_atleta_equipe
  FK id_atleta -> Atleta.id_atleta
  FK id_equipe_campeonato -> Equipe_Campeonato.id_equipe_campeonato
  numero_camisa
  posicao
  data_inicio
  data_fim
  ativo

[Estadio]
  PK id_estadio
  nome
  cidade
  estado
  capacidade
  tipo
  ativo

[Partida]
  PK id_partida
  FK id_campeonato -> Campeonato.id_campeonato
  FK id_equipe_mandante -> Equipe_Campeonato.id_equipe_campeonato
  FK id_equipe_visitante -> Equipe_Campeonato.id_equipe_campeonato
  FK id_estadio -> Estadio.id_estadio
  data_hora
  fase
  rodada
  situacao
  publico_presente
  observacoes

[Resultado_Partida]
  PK id_resultado
  FK id_partida -> Partida.id_partida
  pontos_mandante
  pontos_visitante
  criterio_desempate
  observacoes

[Evento_Partida]
  PK id_evento
  FK id_partida -> Partida.id_partida
  FK id_atleta_equipe -> Atleta_Equipe.id_atleta_equipe
  periodo
  minuto
  segundo
  tipo_evento
  valor_pontos
  descricao

[Pontuacoes_Atletas]
  PK id_pontuacao
  FK id_partida -> Partida.id_partida
  FK id_atleta_equipe -> Atleta_Equipe.id_atleta_equipe
  tipo_pontuacao
  pontos
  minuto
  observacoes

[Arbitro]
  PK id_arbitro
  nome
  documento
  federacao
  federado
  nivel
  ativo

[Arbitro_Partida]
  PK id_arbitro_partida
  FK id_arbitro -> Arbitro.id_arbitro
  FK id_partida -> Partida.id_partida
  funcao
  escala_confirmada
  observacoes
