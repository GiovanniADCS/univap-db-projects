[Modalidade]
  PK id_modalidade
  nome
  tipo_resultado
  descricao
  ativo

[Categoria]
  PK id_categoria
  nome
  idade_minima
  idade_maxima
  descricao
  ativo

[Genero]
  PK id_genero
  sigla
  nome
  descricao
  ativo

[Campeonato]
  PK id_campeonato
  FK id_modalidade -> Modalidade.id_modalidade
  FK id_categoria -> Categoria.id_categoria
  FK id_genero -> Genero.id_genero
  nome
  formato
  situacao
  descricao

[Temporada]
  PK id_temporada
  ano
  nome
  data_inicio
  data_fim
  situacao

[Campeonato_Temporada]
  PK id_campeonato_temporada
  FK id_campeonato -> Campeonato.id_campeonato
  FK id_temporada -> Temporada.id_temporada
  nome_edicao
  data_inicio
  data_fim
  situacao

[Equipe]
  PK id_equipe
  nome
  cidade
  estado
  pais
  data_fundacao
  ativo

[Equipe_Campeonato]
  PK id_equipe_campeonato
  FK id_equipe -> Equipe.id_equipe
  FK id_campeonato_temporada -> Campeonato_Temporada.id_campeonato_temporada
  grupo
  numero_inscricao
  data_inscricao
  situacao

[Atleta]
  PK id_atleta
  FK id_genero -> Genero.id_genero
  nome
  data_nascimento
  nacionalidade
  documento
  ativo

[Atleta_Equipe]
  PK id_atleta_equipe
  FK id_atleta -> Atleta.id_atleta
  FK id_equipe -> Equipe.id_equipe
  numero_camisa
  posicao
  data_inicio
  data_fim
  ativo

[Partida]
  PK id_partida
  FK id_campeonato_temporada -> Campeonato_Temporada.id_campeonato_temporada
  FK id_equipe_mandante -> Equipe_Campeonato.id_equipe_campeonato
  FK id_equipe_visitante -> Equipe_Campeonato.id_equipe_campeonato
  data_hora
  local_partida
  cidade
  estado
  fase
  rodada
  situacao
  publico_presente
  observacoes

[Resultado_Partida]
  PK id_resultado
  FK id_partida -> Partida.id_partida
  pontos_mandante
  pontos_visitante
  criterio_desempate
  observacoes

[Evento_Partida]
  PK id_evento
  FK id_partida -> Partida.id_partida
  FK id_equipe_campeonato -> Equipe_Campeonato.id_equipe_campeonato
  FK id_atleta -> Atleta.id_atleta
  periodo
  minuto
  segundo
  tipo_evento
  valor_pontos
  descricao

[Arbitro]
  PK id_arbitro
  nome
  documento
  federacao
  nivel
  ativo

[Arbitro_Partida]
  PK id_arbitro_partida
  FK id_arbitro -> Arbitro.id_arbitro
  FK id_partida -> Partida.id_partida
  funcao
  escala_confirmada
  observacoes
